	Instructions:
	git clone https://github.com/kujonello79/KJDE.git
	cd KJDE
	cp kjde.tar.gz ~/kjde.tar.gz
	cd
	tar -xvzf kjde.tar.gz
	cd dotfiles/
	sudo sh install.sh

	doas sh install.sh
